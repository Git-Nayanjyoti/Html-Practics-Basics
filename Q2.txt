        function Domain() {
            var domain = document.getElementById("domain").value;
            if (domain == 'jee') {
                document.getElementById('jee').style.display = "block";
                document.getElementById('net').style.display = "none";
            }
            else if (domain == "net") {
                document.getElementById('jee').style.display = "none";
                document.getElementById('net').style.display = "block";
            }
            else {
                document.getElementById('jee').style.display = "none";
                document.getElementById('net').style.display = "none";
            }



        }
        function score() {
            var mpt_marks = document.getElementById('mpt').value;
            var mtt_marks = document.getElementById('mtt').value;
            var assignment_marks = document.getElementById('assignment').value;

            if (isNaN(mpt_marks) || mpt_marks < 1 || mpt_marks > 64) {
                alert("Please enter correct values for MPT marks")
            }
            else if (isNaN(mtt_marks) || mtt_marks < 1 || mtt_marks > 70) {
                alert("Please enter correct values for MTT marks")
            }
            else if (isNaN(assignment_marks) || assignment_marks < 1 || assignment_marks > 100) {
                alert("Please enter correct values for Assignment marks")
            }
            else {
                assignment_marks = assignment_marks * (15 / 100);
                mtt_marks = mtt_marks * (15 / 100);
                mpt_marks = mpt_marks * (70 / 100);

                total_marks = assignment_marks + mtt_marks + mpt_marks;

                if (total_marks <= 49) {
                    document.getElementById('result').innerHTML = '<p class="fail">As your total score is ' + total_marks + ', you have FAILED in your exam with a score of 0 </p>';
                }
                else if (total_marks >= 50 && total_marks <= 59) {
                    document.getElementById('result').innerHTML = '<p class="fail">As your total score is ' + total_marks + ', you have FAILED in your exam with a score of 1</p>';
                }
                else if (total_marks >= 60 && total_marks <= 69) {
                    document.getElementById('result').innerHTML = '<p class="pass">As your total score is ' + total_marks + ', you have PASSED in your exam with a score of 2</p>';
                }
                else if (total_marks >= 70 && total_marks <= 79) {
                    document.getElementById('result').innerHTML = '<p class="pass">As your total score is ' + total_marks + ', you have PASSED in your exam with a score of 3</p>';
                }
                else if (total_marks >= 80 && total_marks <= 89) {
                    document.getElementById('result').innerHTML = '<p class="pass">As your total score is ' + total_marks + ', you have PASSED in your exam with a score of 4</p>';
                }
                else {
                    document.getElementById('result').innerHTML = 'As your total score is ' + total_marks + ', you have PASSED in your exam with a score of 5';
                }
            }
        }
    </script>
